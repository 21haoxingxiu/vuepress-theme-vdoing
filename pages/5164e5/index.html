function _instanceOf(left, right) {
  let leftProto = left.__proto__
  while (true) {
    if (leftProto === right.prototype) {
      return true
    }
    if (leftProto === null) {
      return false
    }
    leftProto = leftProto.__proto__
  }
}

# 实现一个new


function _newFactory(func, ...args) {
  if (typeof func !== 'function') {
    throw new Error('newOperator function the first param must be a function')
  }
  let obj = Object.create(func.prototype)
  let res = func.apply(obj, args)
  isFunc =
    (typeof res === 'Object' || typeof res === 'function') && res !== null
  if (isFunc) {
    return res
  }
  return obj
}

function _newFactory(fn, ...args) {
  if (typeof fn !== 'function') {
    throw new Error('fn is must a function')
  }
  let obj = new Object()
  obj.__proto__ = fn.prototype
  let res = fn.apply(obj, args)
  return typeof res === 'function' || typeof res === 'object' ? res : obj
}

# 防抖 从最后一次触发开始计时（类似回城，打断重新计时）

function debounce(fn, delay) {
  let timer = null
  const last = +Date.now()
  return function () {
    if (timer) clearTimeout(timer)
    timer = setTimeout(() => {
      const time = +Date.now() - last
      console.log('时间间隔', time)
      fn()
    }, delay)
  }
}

# 节流 按照一定频率

function throttle(fn, time) {
  let flag = true
  return function (arg) {
    if (!flag) return
    flag = false
    setTimeout(() => {
      fn(arg)
      flag = true
    }, time)
  }
}


/* 节流2 */
function throttle(fn, time) {
  let last = 0
  return function (arg) {
    let now = Date.now()
    if (now - last >= time) {
      last = now
      fn(arg)
    }
  }
}

# 实现一个bind

一个绑定函数也能使用new操作符创建对象：这种行为就像把原函数当成构造器。提供的 this 值被忽略，同时调用时的参数被提供给模拟函数。

了解 new 的模拟实现，就会知道这个时候的 this 已经指向了 新的创建的 Obj

Function.prototype.bind = function (context = null, ...arg) {
  let tempFn = function() {} // 使用中间函数过渡
  const self = this
  let resFn = function (...args) {
    // 判断是否作为构造函数使用，绑定this，避免this失效
    self.apply(this instanceof resFn ? this : context, arg.concat(args))
  }
  tempFn.prototype = this.prototype
  resFn.prototype = new tempFn()
  return resFn
}

/* 使用 Object.create（） */

  const self = this
  let resFn = function(...args) {
    self.apply(this instanceof resFn ? this : context, arg.concat(args))
  }
  resFn.prototype = Object.create(this.prototype)

# 实现一个call

Function.prototype.call = function (context = window, ...arg) {
  if (typeof this !== 'function') {
    throw new Error('this must is a function')
  }
  context.fn = this
  const result = context.fn(...arg)
  delete context.fn
  return result
}

# 实现一个深拷贝

/* map内是强引用 所占的内存空间不会被释放 map 改成 weakmap 形成弱引用 */
function deepClone(target, map = new weakMap()) {
  if(map.get(target)) return target
  if(!isObject(target)) return  target
  map.set(target, true)
  let cloneTarget = Array.isArray(target) ? [] : {}
  for(let prop in target) {
    if(target.hasOwnProperty(prop)) {
      cloneTarget[prop] = deepClone(target[prop], map)
    }
  }
  return cloneTarget
}

/* 完整版 考虑边界情况 */
const getType = (obj) => Object.prototype.toString.call(obj)

const isObject = (target) =>
  (typeof target === 'object' || typeof target === 'function') &&
  target !== null

const canTraverse = {
  '[object Map]': true,
  '[object Set]': true,
  '[object Array]': true,
  '[object Object]': true,
  '[object Arguments]': true,
}

const mapTag = '[object Map]'
const setTag = '[object Set]'
const boolTag = '[object Boolean]'
const numberTag = '[object Number]'
const stringTag = '[object String]'
const symbolTag = '[object Symbol]'
const dateTag = '[object Date]'
const errorTag = '[object Error]'
const regexpTag = '[object RegExp]'
const funcTag = '[object Function]'

const handleRegExp = (target) => {
  const { source, flags } = target
  return new target.constructor(source, flags)
}

/* 函数处理 */
const handleFunc = (func) => {
  // 箭头函数直接返回自身
  if (!func.prototype) return func
  const bodyReg = /(?<={)(.|\n)+(?=})/m
  const paramReg = /(?<=\().+(?=\)\s+{)/
  const funcString = func.toString()
  // 分别匹配 函数参数和 函数体
  const param = paramReg.exec(funcString)
  const body = bodyReg.exec(funcString)
  if (!body) return null
  if (param) {
    const paramArr = param[0].split(',')
    return new Function(...paramArr, body[0])
  } else {
    return new Function(body[0])
  }
}

/* 不可遍历处理 */
const handleNotTraverse = (target, tag) => {
  const Ctor = target.constructor
  switch (tag) {
    case boolTag:
      return new Object(Boolean.prototype.valueOf.call(target))
    case numberTag:
      return new Object(Number.prototype.valueOf.call(target))
    case stringTag:
      return new Object(String.prototype.valueOf.call(target))
    case symbolTag:
      return new Object(Symbol.prototype.valueOf.call(target))
    case errorTag:
    case dateTag:
      return new Ctor(target)
    case regexpTag:
      return handleRegExp(target)
    case funcTag:
      return handleFunc(target)
    default:
      return new Ctor(target)
  }
}

const deepClone = (target, map = new weakMap()) => {
  if (!isObject(target)) return target
  let type = getType(target)
  let cloneTarget
  if (!canTraverse[type]) {
    // 处理不能遍历的对象
    // return handleNotTraverse(target, type)
    /* 处理日期和正则 */
    if (type === '[object Date]') {
      return new target.contructor(target)
    }
    if (type === '[object RegExp]') {
      const { source, flags } = target
      return new target.constructor(source, flags)
    }
  } else {
    // 这波操作相当关键， 可以保证对象的原型不丢失
    let ctor = target.constructor
    cloneTarget = new ctor()
  }

  if (map.get(target)) return target
  map.set(target, true)

  if (type === mapTag) {
    // 处理map
    target.forEach((item, key) => {
      /* 属性和内容都深拷贝 */
      cloneTarget.set(deepClone(key, map), deepClone(item, map))
    })
  }

  if (type === setTag) {
    // 处理set
    target.forEach((item) => {
      cloneTarget.add(deepClone(item, map))
    })
  }

  //处理数组和对象
  for (let prop in target) {
    if (target.hasOwnProperty(prop)) {
      cloneTarget[prop] = deepClone(target[prop], map)
    }
  }

  return cloneTarget
}

# 实现一个vue自定义指令

/* 【高频】实现一个 Vue 自定义指令懒加载 */
;<img v-lazyload="http:323432.jpg" />

vue.directive('lazyload', {
  // 指令的定义
  bind: function (el, binding) {
    let lazyimageObserver = new IntersectionObserver((entries, observer) => {
      entries.forEach((entry, index) => {
        let lazyImage = entry.target
        // 相交率
        if (entry.intersectionRatio > 0) {
          lazyImage.src = binding.value
          /* 当前图片加载完之后需要去掉监听 */
          lazyimageObserver.unobserve(lazyImage)
        }
      })
    })
    lazyimageObserver.observe(el)
  },
})

# 观察者模式 (产品发布需求)

class Publisher {
  constructor() {
    this.observers = []
  }

  add(observer) {
    this.observers.push(observer)
  }

  remove(observer) {
    this.observers.forEach((item, index) => {
      if(item === observer) {
        this.observers.splice(index, 1)
      }
    })
  }

  notify() {
    this.observers.forEach(observer => {
      observer.update(this)
    })
  }
}

class Observer {
  constructor() {

  }
  update() {

  }
}

class ProdPublisher extends Publisher {
  constructor() {
    super()
    this.prodState = null
  }

  getState() {
    return this.prodState
  }

  setState(state) {
    this.prodState = state
    this.notify()
  }
}

class ProdObserver extends Observer {
  constructor() {
    super()
    this.prodState = null
  }

  update(publisher) {
    this.prodState = publisher.getState()
    this.work()
  }

  work() {
    console.log('prod', this.prodState);
    console.log('996 开始, working');
  }
}


var prod = {

}

var qd = new ProdObserver()
var hd = new ProdObserver()

/* 收集订阅者 */
cp.add(qd)
cp.add(hd)

/* 需求来了  working */
cp.setState(prod)

# 实现一个event bus （发布-订阅模式）

class EventEmitter {
  constructor() {
    // 用于存储事件与回调之间的对应关系
    this.handles = {}
  }

  on(eventName, cb) {
    if(!this.handles[eventName]) {
      this.handles[eventName] = []
    }

    this.handles[eventName].push(cb)
  }

  emit(eventName, ...args) {
    if(this.handles[eventName]) {
      this.handles[eventName].forEach(callback => {
        callback(...args)
      })
    }
  }

  off(eventName, cb) {
    const callback = this.handles[eventName]
    const index = callback.indexOf(cb)
    if(index !== -1) {
      callback.splice(index, 1)
    }
  }

  once(eventName, cb){
    const wrapper = (...args) => {
      cb(...args)
      this.off(eventName, wrapper)
    }
    this.on(eventName, wrapper)
  }
}

/* 20行 实现promise */

function _Promise(resolve) {
  this.data = null
  const resolve = (value) => {
    setTimeout(() => {
      this.data = value
      this.cbs.forEach(cb => {
        cb()
      })      
    });

  }

  fn(resolve)
}

_Promise.prototype.then = (onResolved) => {
  this.cbs = []
  return new _Promise((resolve) => {
    this.cbs.push(() => { // 考虑到 promise 实例多少 then的情况
      const res = onResolved(this.data)
      if(res instanceof _Promise) {
        res.then(resolve)
      } else {
        resolve(res)
      }
    })
  })
}
